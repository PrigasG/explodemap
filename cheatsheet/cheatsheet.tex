\shead{Displacement Model}

\vspace{-2pt}
\[
t_i = \alpha_r\,\hat{d}_{\text{state},i} + \alpha_l\,s_i\,\hat{d}_{\text{local},i}
\]
\vspace{-6pt}

{\scriptsize
\begin{tabular}{@{}ll@{}}
$\hat{d}_{\text{state},i}$ & $C_s \rightarrow C_r$ (regional push) \\
$\hat{d}_{\text{local},i}$ & $C_r \rightarrow C_i$ (local push) \\
$s_i = (r_i / d_{\max})^p$ & 0 at center, 1 at edge \\
\end{tabular}
}

\shead{Parameter Derivation}

\sshead{Regional separation}
\vspace{-3pt}
\[
\alpha_r = \frac{\gamma_r \bar{w}}{2\sin(\pi / n_r)}
\]

\sshead{Local expansion}
\vspace{-6pt}
\[
\alpha_l = \gamma_l \frac{2\,\widetilde{R}_{\text{local}}}{\sqrt{\bar{n}}}
\]
\vspace{-6pt}

{\scriptsize
\begin{tabular}{@{}lll@{}}
$\bar{w}$ & Median polygon diameter & \\
$\widetilde{R}_{\text{local}}$ & Median regional radius & \\
$\bar{n}$ & Median units/region & \\
$n_r$ & Number of regions & \\
$\gamma_r$ & Regional clearance & 3.0 \\
$\gamma_l$ & Local clearance & 1.136 \\
$p$ & Distance exponent & 1.25 \\
\end{tabular}
}

\smallskip
\fn{compute\_stats}(\parg{sf\_obj}, \parg{region\_col})\\
\fn{derive\_params}(\parg{stats}, \parg{gamma\_r}, \parg{gamma\_l})

\shead{Guarantees (2-Level Core)}

\sshead{Proposition 1}
Rigid translation: shape, area, perimeter, and topology preserved.

\sshead{Proposition 2}
Radial order within each region is preserved.

\sshead{Proposition 3}
\[
\|t_i\| \le \alpha_r + \alpha_l
\]
